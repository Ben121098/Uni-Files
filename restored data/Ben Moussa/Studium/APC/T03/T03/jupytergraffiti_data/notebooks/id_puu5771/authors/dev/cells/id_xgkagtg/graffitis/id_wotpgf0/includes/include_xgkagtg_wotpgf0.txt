# Defining number of steps
N_sim = 60

# Saving the data
x_data = [x0]
x_hat_data = [x0_observer]

# Luenberger Observer Loop
for i in range(N_sim):
    y = C @ x0
    x0 = A @ x0 + B @ u
    x_data.append(x0)

    y_hat = C @ x0_observer
    x0_observer = A @ x0_observer + B @ u + L @ (y - y_hat)
    x_hat_data.append(x0_observer)
